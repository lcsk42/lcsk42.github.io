function ppgrep() {
    if [[ $1 == "" ]]; then
        FZF=fzf
    else
        FZF="fzf -q $1"
    fi
    ps aux | eval $FZF | awk '{ print $2 }'
}

function ppkill() {
    if [[ $1 =~ "^-" ]]; then
        QUERY=""            # options only
    else
        QUERY=$1            # with a query
        [[ $# > 0 ]] && shift
    fi
    ppgrep $QUERY | xargs kill $*
}
